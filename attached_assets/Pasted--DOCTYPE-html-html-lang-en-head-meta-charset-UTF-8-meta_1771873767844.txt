        // Form submission handling with progress bar
        const form = document.querySelector('form');
        const evaluateBtn = document.getElementById('evaluateBtn');
        const evaluateIcon = document.getElementById('evaluateIcon');
        const evaluateText = document.getElementById('evaluateText');
        const progressContainer = document.getElementById('progressContainer');
        const progressFill = document.getElementById('progressFill');
        const progressText = document.getElementById('progressText');
        const progressDetails = document.getElementById('progressDetails');

        function resetButton() {
            evaluateBtn.disabled = false;
            evaluateIcon.className = 'fas fa-check-circle';
            evaluateText.textContent = 'Evaluate Papers';
            progressContainer.style.display = 'none';
        }

        let progressEventSource = null;

        form.addEventListener('submit', function(e) {
            e.preventDefault(); // Prevent form navigation
            
            // Validate files are selected
            const questionPaper = document.getElementById('question_paper').files[0];
            const answerPapers = document.getElementById('answer_papers').files;
            
            // Additional Instruction Set is optional, so no validation needed for questionPaper
            
            if (!answerPapers || answerPapers.length === 0) {
                alert('Please select at least one document to process.');
                resetButton();
                return;
            }

            // Disable button and show progress
            evaluateBtn.disabled = true;
            evaluateIcon.className = 'loading-spinner';
            evaluateText.textContent = 'Processing...';
            progressContainer.style.display = 'block';
            
            // Initialize progress
            progressFill.style.width = '0%';
            progressText.textContent = 'Initializing...';
            progressDetails.textContent = 'Starting evaluation process...';
            
            // Start the async evaluation process
            startEvaluationProcess();
        });

        async function startEvaluationProcess() {
            try {
                // Step 1: Start evaluation to get progress ID
                const startResponse = await fetch('/start-evaluation', {
                    method: 'POST',
                    headers: {
                        'X-Requested-With': 'XMLHttpRequest'
                    }
                });
                
                if (!startResponse.ok) {
                    throw new Error('Failed to start evaluation');
                }
                
                const startData = await startResponse.json();
                if (startData.error) {
                    throw new Error(startData.error);
                }
                
                // Step 2: Start progress tracking with the progress ID
                startProgressTracking(startData.progress_id);
                
                // Step 3: Send files for processing
                const formData = new FormData(form);
                const processResponse = await fetch('/evaluate', {
                    method: 'POST',
                    body: formData,
                    headers: {
                        'X-Requested-With': 'XMLHttpRequest'
                    }
                });
                
                const processData = await processResponse.json();
                
                if (processData.error) {
                    throw new Error(processData.error);
                }
                
                // Success - wait for progress to complete, then redirect
                setTimeout(() => {
                    window.location.href = '/view-results';
                }, 2000);
                
            } catch (error) {
                console.error('Evaluation error:', error);
                handleProgressError(error.message);
            }
        }
        
        function startProgressTracking(progressId) {
            // Use EventSource (SSE) if available, fallback to polling
            if (typeof EventSource !== 'undefined') {
                startSSEProgress(progressId);
            } else {
                pollProgress(progressId);
            }
        }
        
        function startSSEProgress(progressId) {
            const eventSource = new EventSource(`/progress/${progressId}`);
            
            eventSource.onmessage = function(event) {
                try {
                    const data = JSON.parse(event.data);
                    updateProgressDisplay(data);
                    
                    if (data.completed || data.error) {
                        eventSource.close();
                        if (data.error) {
                            handleProgressError(data.error);
                        }
                    }
                } catch (error) {
                    console.error('Error parsing SSE data:', error);
                }
            };
            
            eventSource.onerror = function(event) {
                console.warn('SSE connection error, falling back to polling');
                eventSource.close();
                pollProgress(progressId);
            };
            
            // Store reference for cleanup
            window.currentEventSource = eventSource;
        }

        function pollProgress(progressId) {
            // Simple polling approach - check for progress every 500ms
            const pollInterval = setInterval(() => {
                const url = progressId ? `/api/progress?progress_id=${progressId}` : '/api/progress';
                fetch(url, {
                    method: 'GET',
                    headers: {
                        'X-Requested-With': 'XMLHttpRequest'
                    }
                })
                .then(response => {
                    if (!response.ok) {
                        throw new Error('Network response was not ok');
                    }
                    return response.json();
                })
                .then(data => {
                    updateProgressDisplay(data);
                    
                    if (data.completed) {
                        clearInterval(pollInterval);
                        if (data.error) {
                            handleProgressError(data.error);
                        }
                        // Don't auto-redirect here, let startEvaluationProcess handle it
                    }
                })
                .catch(error => {
                    console.error('Progress polling error:', error);
                    // Continue polling, don't stop on temporary errors
                });
            }, 500);
            
            // Stop polling after 5 minutes as failsafe
            setTimeout(() => {
                clearInterval(pollInterval);
            }, 300000);
            
            // Store interval so we can clear it if needed
            window.currentPollInterval = pollInterval;
        }

        function updateProgressDisplay(data) {
            if (data.percentage !== undefined) {
                progressFill.style.width = data.percentage + '%';
            }
            if (data.message) {
                progressText.textContent = data.message;
            }
            if (data.status) {
                const statusMessages = {
                    'uploading': 'Uploading files...',
                    'extracting': 'Extracting text from documents...',
                    'processing': 'Processing files...',
                    'evaluating': 'AI evaluation in progress...',
                    'saving': 'Generating reports...',
                    'finished': 'Processing complete!'
                };
                progressDetails.textContent = statusMessages[data.status] || data.status;
            }
        }

        function handleProgressError(error) {
            progressText.textContent = 'Error occurred';
            progressDetails.textContent = error;
            progressFill.style.width = '100%';
            progressFill.style.backgroundColor = '#dc3545'; // Red color for error
            resetButton();
        }

        const modelSelect = document.getElementById('model');
        const modelOptions = document.querySelectorAll('.model-option');

        // Initialize with selected model
        const selectedOption = document.querySelector('.model-option.selected');
        if (selectedOption) {
            modelSearch.value = selectedOption.textContent;
        }

        modelSearch.addEventListener('input', function() {
            const searchValue = this.value.toLowerCase();

            modelOptions.forEach(option => {
                const optionText = option.textContent.toLowerCase();
                if (optionText.includes(searchValue)) {
                    option.style.display = 'block';
                } else {
                    option.style.display = 'none';
                }
            });
        });

        modelOptions.forEach(option => {
            option.addEventListener('click', function() {
                modelOptions.forEach(opt => opt.classList.remove('selected'));
                this.classList.add('selected');

                const selectedValue = this.getAttribute('data-value');
                modelSelect.value = selectedValue;
                modelSearch.value = this.textContent;
            });
        });

        // File upload handling
        const questionUpload = document.getElementById('questionUpload');
        const questionInput = document.getElementById('question_paper');
        const questionFileInfo = document.getElementById('questionFileInfo');

        const answerUpload = document.getElementById('answerUpload');
        const answerInput = document.getElementById('answer_papers');
        const answerFileInfo = document.getElementById('answerFileInfo');

        questionUpload.addEventListener('click', () => {
            questionInput.click();
        });

        questionInput.addEventListener('change', function() {
            if (this.files.length > 0) {
                questionFileInfo.innerHTML = '';
                const fileTag = document.createElement('div');
                fileTag.className = 'file-tag';
                fileTag.innerHTML = this.files[0].name + '<i class="fas fa-times"></i>';

                const removeIcon = fileTag.querySelector('i');
                removeIcon.addEventListener('click', (e) => {
                    e.stopPropagation();
                    questionInput.value = '';
                    questionFileInfo.innerHTML = '';
                });

                questionFileInfo.appendChild(fileTag);
            }
        });

        answerUpload.addEventListener('click', () => {
            answerInput.click();
        });

        answerInput.addEventListener('change', function() {
            if (this.files.length > 0) {
                answerFileInfo.innerHTML = '';

                for (let i = 0; i < this.files.length; i++) {
                    const fileTag = document.createElement('div');
                    fileTag.className = 'file-tag';
                    fileTag.innerHTML = this.files[i].name;
                    answerFileInfo.appendChild(fileTag);
                }

                if (this.files.length > 1) {
                    const countTag = document.createElement('div');
                    countTag.className = 'file-tag';
                    countTag.innerHTML = `${this.files.length} files selected`;
                    answerFileInfo.appendChild(countTag);
                }

        });

        // Prompt Management System
        const savePromptBtn = document.getElementById('savePromptBtn');
        const editPromptBtn = document.getElementById('editPromptBtn');
        const deletePromptBtn = document.getElementById('deletePromptBtn');
        const savedPromptsDropdown = document.getElementById('savedPrompts');
        const systemPromptTextarea = document.getElementById('system_prompt');
        const templateSelect = document.getElementById('template_name');
        const savePromptText = document.getElementById('savePromptText');
        
        let editMode = false;
        let currentEditingPrompt = null;

        // Show/hide Edit and Delete buttons based on selection
        function togglePromptButtons() {
            const hasSelection = savedPromptsDropdown.value !== '';
            editPromptBtn.style.display = hasSelection ? 'inline-block' : 'none';
            deletePromptBtn.style.display = hasSelection ? 'inline-block' : 'none';
        }

        // Handle saved prompt selection
        savedPromptsDropdown.addEventListener('change', function() {
            const selectedPromptName = this.value;
            togglePromptButtons();
            
            if (!selectedPromptName) {
                exitEditMode();
                return; // Empty selection
            }
            
            // Load the selected prompt
            loadPrompt(selectedPromptName);
        });

        // Handle save prompt button click (Save or Update)
        savePromptBtn.addEventListener('click', function() {
            const promptText = systemPromptTextarea.value.trim();
            const selectedTemplate = templateSelect.value;
            
            if (!promptText) {
                alert('Please enter some prompt text before saving.');
                return;
            }
            
            if (editMode) {
                // Update existing prompt
                updatePrompt(currentEditingPrompt, promptText, selectedTemplate);
            } else {
                // Create new prompt
                const promptName = prompt('Enter a friendly name for this prompt:\n(This name will help you identify the prompt later)', '');
                
                if (!promptName || !promptName.trim()) {
                    return; // User cancelled or entered empty name
                }
                
                savePrompt(promptName.trim(), promptText, selectedTemplate);
            }
        });

        // Handle edit button click
        editPromptBtn.addEventListener('click', function() {
            const selectedPromptName = savedPromptsDropdown.value;
            if (!selectedPromptName) return;
            
            enterEditMode(selectedPromptName);
        });

        // Handle delete button click
        deletePromptBtn.addEventListener('click', function() {
            const selectedPromptName = savedPromptsDropdown.value;
            if (!selectedPromptName) return;
            
            if (confirm(`Are you sure you want to delete the prompt "${selectedPromptName}"?\n\nThis action cannot be undone.`)) {
                deletePrompt(selectedPromptName);
            }
        });

        // Enter edit mode
        function enterEditMode(promptName) {
            editMode = true;
            currentEditingPrompt = promptName;
            savePromptText.textContent = 'Update';
            savePromptBtn.style.background = 'var(--info-color)';
            
            // Visual feedback
            const feedback = document.createElement('div');
            feedback.id = 'editModeIndicator';
            feedback.style.cssText = 'position: fixed; top: 20px; right: 20px; background: #17a2b8; color: white; padding: 10px 20px; border-radius: 4px; z-index: 1000; font-size: 14px;';
            feedback.innerHTML = '<i class="fas fa-edit"></i> Edit Mode: ' + promptName;
            document.body.appendChild(feedback);
        }

        // Exit edit mode
        function exitEditMode() {
            editMode = false;
            currentEditingPrompt = null;
            savePromptText.textContent = 'Save';
            savePromptBtn.style.background = 'var(--primary-color)';
            
            // Remove edit mode indicator
            const indicator = document.getElementById('editModeIndicator');
            if (indicator) {
                document.body.removeChild(indicator);
            }
        }

        // Save prompt function
        async function savePrompt(name, text, templateName) {
            try {
                const response = await fetch('/save-prompt', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({
                        prompt_name: name,
                        prompt_text: text,
                        template_name: templateName
                    })
                });
                
                const data = await response.json();
                
                if (data.success) {
                    alert('Prompt saved successfully!');
                    // Add the new prompt to the dropdown
                    const newOption = document.createElement('option');
                    newOption.value = name;
                    newOption.textContent = name;
                    savedPromptsDropdown.appendChild(newOption);
                    savedPromptsDropdown.value = name;
                    togglePromptButtons();
                } else {
                    alert('Error saving prompt: ' + data.message);
                }
            } catch (error) {
                console.error('Save prompt error:', error);
                alert('Error saving prompt. Please try again.');
            }
        }

        // Update prompt function
        async function updatePrompt(name, text, templateName) {
            try {
                // Delete old prompt and create new one with same name
                await fetch(`/delete-prompt/${encodeURIComponent(name)}`, {
                    method: 'POST'
                });
                
                const response = await fetch('/save-prompt', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({
                        prompt_name: name,
                        prompt_text: text,
                        template_name: templateName
                    })
                });
                
                const data = await response.json();
                
                if (data.success) {
                    alert('Prompt updated successfully!');
                    exitEditMode();
                } else {
                    alert('Error updating prompt: ' + data.message);
                }
            } catch (error) {
                console.error('Update prompt error:', error);
                alert('Error updating prompt. Please try again.');
            }
        }

        // Delete prompt function
        async function deletePrompt(name) {
            try {
                const response = await fetch(`/delete-prompt/${encodeURIComponent(name)}`, {
                    method: 'POST'
                });
                
                const data = await response.json();
                
                if (data.success) {
                    alert('Prompt deleted successfully!');
                    // Remove from dropdown
                    const optionToRemove = Array.from(savedPromptsDropdown.options).find(
                        option => option.value === name
                    );
                    if (optionToRemove) {
                        savedPromptsDropdown.removeChild(optionToRemove);
                    }
                    savedPromptsDropdown.value = '';
                    togglePromptButtons();
                    exitEditMode();
                    
                    // Clear the form
                    systemPromptTextarea.value = 'You are a helpful assistant that analyzes and processes documents. Provide detailed analysis, insights, and structured feedback based on the content provided. Focus on key findings, recommendations, and actionable insights. Maintain a professional and clear tone throughout your analysis. Structure your output in a clear, organized manner that highlights the most important points and provides valuable insights for the reader.';
                    templateSelect.value = '';
                } else {
                    alert('Error deleting prompt: ' + data.message);
                }
            } catch (error) {
                console.error('Delete prompt error:', error);
                alert('Error deleting prompt. Please try again.');
            }
        }

        // Initialize button visibility
        togglePromptButtons();

        // Load prompt function
        async function loadPrompt(promptName) {
            try {
                const response = await fetch(`/load-prompt/${encodeURIComponent(promptName)}`);
                const data = await response.json();
                
                if (data.success) {
                    const promptData = data.prompt;
                    
                    // Set the prompt text
                    systemPromptTextarea.value = promptData.text;
                    
                    // Auto-select the associated template if it exists
                    if (promptData.template) {
                        const templateOption = Array.from(templateSelect.options).find(
                            option => option.value === promptData.template
                        );
                        if (templateOption) {
                            templateSelect.value = promptData.template;
                        }
                    }
                    
                    // Visual feedback
                    const feedback = document.createElement('div');
                    feedback.style.cssText = 'position: fixed; top: 20px; right: 20px; background: #28a745; color: white; padding: 10px 20px; border-radius: 4px; z-index: 1000; font-size: 14px;';
                    feedback.textContent = `Loaded prompt: "${promptName}"`;
                    if (promptData.template) {
                        feedback.textContent += ` with template: "${promptData.template}"`;
                    }
                    document.body.appendChild(feedback);
                    
                    // Remove feedback after 3 seconds
                    setTimeout(() => {
                        if (document.body.contains(feedback)) {
                            document.body.removeChild(feedback);
                        }
                    }, 3000);
                } else {
                    alert('Error loading prompt: ' + data.message);
                }
            } catch (error) {
                console.error('Load prompt error:', error);
                alert('Error loading prompt. Please try again.');
            }
        }
    </script>
